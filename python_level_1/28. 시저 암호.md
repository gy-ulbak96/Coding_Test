```python
def solution(s, n):
    a = ''
    upper=list(map(lambda x : chr(x), range(65,91)))
    lower = list(str(i).lower() for i in upper)
    for i in s:
        if i ==' ':
            a += ' '
        elif i.islower():
            a += lower[((lower.index(i))+n)%26]
        else:
            a += upper[((upper.index(i))+n)%26]
        
    return a
```

```python
def caesar(s, n):
```

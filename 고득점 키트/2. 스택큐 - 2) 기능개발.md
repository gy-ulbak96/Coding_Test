def solution(progresses, speeds):
    result=[]
    work = list(map(lambda x,y:(100-x)//y if (100-x)%y==0 else (100-x)//y+1,progresses,speeds))
    print(work)
    work = deque(work)
    
    while work:
        r = work.popleft()
        count = 1
        work1 = work
        for i in work1:
            if i <= r:
                print(i,r)
                count += 1
            else:
                print(work)
                break
        for i in range(count-1):
            work.popleft()
        result.append(count)     
    return result

def solution(p):
    ans = [0] * len(p)
    stack = [0]
    for i in range(1, len(p)):
        if p[i] < p[stack[-1]]:
            for j in stack[::-1]:
                if p[i] < p[j]:
                    ans[j] = i-j
                    stack.remove(j)
                else:
                    break
        stack.append(i)
    for i in range(0, len(stack)-1):
        ans[stack[i]] = len(p) - stack[i] - 1
    return ans

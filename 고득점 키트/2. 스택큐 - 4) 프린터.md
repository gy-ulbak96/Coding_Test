```python
def solution(p, l):
    ans = 0
    m = max(p)
    while True:
        v = p.pop(0)
        if m == v:
            ans += 1
            if l == 0:
                break
            else:
                l -= 1
            m = max(p)
        else:
            p.append(v)
            if l == 0:
                l = len(p)-1
            else:
                l -= 1
    return ans
```

```python
def solution(priorities, location):
    queue =  [(i,p) for i,p in enumerate(priorities)]

    while True:
        cur = queue.pop(0)
        if any(cur[1] < q[1] for q in queue):

            answer += 1
            if cur[0] == location:
                return answer
```

import heapq

def solution(operations):
    result = []
    for order in operations:
        a,b = order.split(" ")
        if a == 'I':
            heapq.heappush(result,int(b))
        else:
            if int(b) < 0:
                heapq.heappop(result)
            else:
                if result != []:
                    result.pop()
    if result == []:
        return [0,0]
    return [max(result),min(result)]
